TOTAL ACTIVE DUMs 
WITH dns_active AS (
  SELECT dl.domain_id
  FROM public.domain_locations dl
  WHERE dl.domain_id IN (SELECT domain_id FROM public.domain_locations WHERE registry = 'DNS')
    AND dl.registry IN ('DNS', 'UNS')
  GROUP BY dl.domain_id
  HAVING MAX(dl.expires_at)::date >= CURRENT_DATE
),
transfer_ins AS (SELECT DISTINCT domain_id FROM public.products WHERE type = 'Web2TransferInProduct')
SELECT COUNT(*), COUNT(*) FILTER (WHERE ti.domain_id IS NULL), COUNT(*) FILTER (WHERE ti.domain_id IS NOT NULL)
FROM dns_active da LEFT JOIN transfer_ins ti ON ti.domain_id = da.domain_id
-- +7d: filtered to MIN(created_at) >= CURRENT_DATE - INTERVAL '7 days'

NEW DUMs — MONTH OVER MONTH 
WITH dns_active AS (
  SELECT dl.domain_id, MIN(dl.created_at) AS first_created
  FROM public.domain_locations dl
  WHERE dl.domain_id IN (SELECT domain_id FROM public.domain_locations WHERE registry = 'DNS')
    AND dl.registry IN ('DNS', 'UNS')
  GROUP BY dl.domain_id
  HAVING MAX(dl.expires_at)::date >= CURRENT_DATE
),
transfer_ins AS (SELECT DISTINCT domain_id FROM public.products WHERE type = 'Web2TransferInProduct')
SELECT DATE_TRUNC('month', da.first_created) AS month,
  COUNT(*) AS new_total,
  COUNT(*) FILTER (WHERE ti.domain_id IS NOT NULL) AS new_transfers
FROM dns_active da LEFT JOIN transfer_ins ti ON ti.domain_id = da.domain_id
WHERE da.first_created >= DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '11 months'
GROUP BY 1 ORDER BY 1
